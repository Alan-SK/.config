113 func! CompileRunText()
114         exec "w"
115         if &filetype == 'python'
116                 set splitbelow
117                 :sp
118                 :term python3 %
119         elseif &filetype == 'sh'
120                 :!time bash %
121         elseif &filetype == 'go'
122                 set splitbelow
123                 :sp
124                 :term go run %
125         endif
126 endfunction
